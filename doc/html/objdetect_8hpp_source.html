00158 // DataType: STRUCT filterObject
00159 // Description of the filter, which corresponds to the part of the object
00160 // V               - ideal (penalty = 0) position of the partial filter
00161 //                   from the root filter position (V_i in the paper)
00162 // penaltyFunction - vector describes penalty function (d_i in the paper)
00163 //                   pf[0] * x + pf[1] * y + pf[2] * x^2 + pf[3] * y^2
00164 // FILTER DESCRIPTION
00165 //   Rectangular map (sizeX x sizeY), 
00166 //   every cell stores feature vector (dimension = p)
00167 // H               - matrix of feature vectors
00168 //                   to set and get feature vectors (i,j) 
00169 //                   used formula H[(j * sizeX + i) * p + k], where
00170 //                   k - component of feature vector in cell (i, j)
00171 // END OF FILTER DESCRIPTION
00172 // xp              - auxillary parameter for internal use
00173 //                   size of row in feature vectors 
00174 //                   (yp = (int) (p / xp); p = xp * yp)
00175 typedef struct{
00176     CvLSVMFilterPosition V;
00177     float fineFunction[4];
00178     unsigned int sizeX;
00179     unsigned int sizeY;
00180     unsigned int p;
00181     unsigned int xp;
00182     float *H;
00183 } CvLSVMFilterObject;
